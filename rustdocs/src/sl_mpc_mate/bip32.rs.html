16/// 4-byte Key fingerprint
17pub type KeyFingerPrint = [u8; 4];

19const KEY_SIZE: usize = 32;

21/// BIP32 version bytes
22#[derive(Debug, Clone, Copy)]
23pub enum Prefix {
24    /// 'xpub' prefix
25    XPub,
26    /// 'ypub' prefix
27    YPub,
28    /// 'zpub' prefix
29    ZPub,
30    /// 'tpub' prefix
31    TPub,
32    /// Custom prefix
33    Custom(u32),
34}
35
36impl Prefix {
37    /// Get the prefix as a 4-byte array
38    pub fn as_bytes(&self) -> [u8; 4] {
39        u32::from(*self).to_be_bytes()
40    }
41}
42
43impl From<Prefix> for u32 {
44    fn from(prefix: Prefix) -> Self {
45        match prefix {
46            Prefix::XPub => 0x0488b21e,
47            Prefix::YPub => 0x049d7cb2,
48            Prefix::ZPub => 0x04b24746,
49            Prefix::TPub => 0x043587cf,
50            Prefix::Custom(prefix) => prefix,
51        }
52    }
53}
54
55impl From<[u8; 4]> for Prefix {
56    fn from(prefix: [u8; 4]) -> Self {
57        match prefix {
58            [0x04, 0x88, 0xb2, 0x1e] => Prefix::XPub,
59            [0x04, 0x9d, 0x7c, 0xb2] => Prefix::YPub,
60            [0x04, 0xb2, 0x47, 0x46] => Prefix::ZPub,
61            [0x04, 0x35, 0x87, 0xcf] => Prefix::TPub,
62            _ => Prefix::Custom(u32::from_be_bytes(prefix)),
63        }
64    }
65}

67/// Extended public key
68#[derive(Clone, Debug)]
69pub struct XPubKey {
70    /// Prefix (version) as a 4-byte integer
71    pub prefix: Prefix,
72    /// Parent fingerprint
73    pub parent_fingerprint: KeyFingerPrint,
74    /// Child number
75    pub child_number: u32,
76    /// Public key
77    pub pubkey: ProjectivePoint,
78    /// Root chain code
79    pub chain_code: [u8; 32],
80    /// Depth
81    pub depth: u8,
82}
83
84fn base58_encode(serialized: [u8; 78]) -> String {
85    let checksum: [u8; 4] = Sha256::digest(Sha256::digest(serialized))[..4]
86        .try_into()
87        .unwrap();
88
89    let serialized = [&serialized, checksum.as_slice()].concat();
90    bs58::encode(serialized)
91        .with_alphabet(Alphabet::BITCOIN)
92        .into_string()
93}

95/// Errors while performing keygen
96#[derive(Error, Debug)]
97pub enum BIP32Error {
98    /// Hardened child index is not supported
99    #[error("Hardened child index is not supported (yet)")]
100    HardenedChildNotSupported,
101    /// Invalid chain code
102    #[error("Invalid chain code")]
103    InvalidChainCode,
104    /// Invalid public key, it cannot be the point at infinity
105    #[error("Invalid public key, cannot be the point at infinity")]
106    PubkeyPointAtInfinity,
107    /// Invalid child key, it cannot be greater than the group order (extremely unlikely)
108    #[error("Invalid child key, cannot be greater than the group order")]
109    InvalidChildScalar,
110}
111
112impl XPubKey {
113    /// Serialize to string
114    /// # Arguments
115    /// * `encoded` - If true, the string will be encoded to Base58 with checksum (like Bitcoin addresses)
116    pub fn to_string(&self, encoded: bool) -> String {
117        let serialized = [
118            &self.prefix.as_bytes(),
119            self.depth.to_be_bytes().as_slice(),
120            &self.parent_fingerprint,
121            &self.child_number.to_be_bytes(),
122            self.chain_code.as_slice(),
123            self.pubkey.to_encoded_point(true).as_bytes(),
124        ]
125        .concat();
126
127        let serialized: [u8; 78] = serialized.try_into().expect(
128            "Invalid serialized extended public key length, must be 78 bytes",
129        );
130
131        if encoded {
132            base58_encode(serialized)
133        } else {
134            hex::encode(serialized)
135        }
136    }
137}

139/// Derive a child public key from a parent public key and a parent chain code
140pub fn derive_child_pubkey(
141    parent_pubkey: &ProjectivePoint,
142    parent_chain_code: [u8; 32],
143    child_number: &ChildIndex,
144) -> Result<(Scalar, ProjectivePoint, [u8; 32]), BIP32Error> {
145    let mut hmac_hasher =
146        Hmac::<sha2::Sha512>::new_from_slice(&parent_chain_code)
147            .map_err(|_| BIP32Error::InvalidChainCode)?;
148
149    if child_number.is_normal() {
150        hmac_hasher.update(parent_pubkey.to_encoded_point(true).as_bytes());
151    } else {
152        return Err(BIP32Error::HardenedChildNotSupported);
153    }
154
155    hmac_hasher.update(&child_number.to_bits().to_be_bytes());
156    let result = hmac_hasher.finalize().into_bytes();
157    let (il_int, child_chain_code) = result.split_at(KEY_SIZE);
158    let il_int = U256::from_be_slice(il_int);
159
160    // Has a chance of 1 in 2^127
161    if il_int > Secp256k1::ORDER {
162        return Err(BIP32Error::InvalidChildScalar);
163    }
164
165    let pubkey = ProjectivePoint::GENERATOR * Scalar::reduce(il_int);
166
167    let child_pubkey = pubkey + parent_pubkey;
168
169    // Return error if child pubkey is the point at infinity
170    if child_pubkey == ProjectivePoint::IDENTITY {
171        return Err(BIP32Error::PubkeyPointAtInfinity);
172    }
173
174    Ok((
175        Scalar::reduce(il_int),
176        child_pubkey,
177        child_chain_code.try_into().unwrap(),
178    ))
179}

181/// Get the fingerprint of the root public key
182pub fn get_finger_print(public_key: &ProjectivePoint) -> KeyFingerPrint {
183    let pubkey_bytes: [u8; 33] = public_key
184        .to_encoded_point(true)
185        .as_bytes()
186        .as_ref()
187        .try_into()
188        .expect("compressed pubkey must be 33 bytes");
189
190    let digest = Ripemd160::digest(Sha256::digest(pubkey_bytes));
191    digest[..4].try_into().expect("digest truncated")
192}

194/// Generate a key ID from a root public key and root chain code
195/// # Arguments
196/// * `root_pubkey` - Root public key SEC1's compressed form (33 bytes)
197/// * `root_chain_code` - Root chain code (32 bytes)
198/// # Returns
199/// * `key_id` - Base64 encoded string
200pub fn generate_key_id(
201    root_pubkey: &ProjectivePoint,
202    root_chain_code: [u8; 32],
203) -> String {
204    let id = sha2::Sha256::new()
205        .chain_update(root_pubkey.to_encoded_point(true).as_bytes().as_ref())
206        .chain_update(root_chain_code)
207        .finalize();
208
209    general_purpose::STANDARD_NO_PAD.encode(id)
210}

212/// Derive the extended public key for a given derivation path and prefix
213/// # Arguments
214/// * `prefix` - Prefix for the extended public key (`Prefix` has commonly used prefixes)
215/// * `chain_path` - Derivation path
216///
217/// # Returns
218/// * `XPubKey` - Extended public key
219pub fn derive_xpub(
220    prefix: Prefix,
221    root_public_key: &ProjectivePoint,
222    root_chain_code: [u8; 32],
223    chain_path: DerivationPath,
224) -> Result<XPubKey, BIP32Error> {
225    let mut pubkey = *root_public_key;
226    let mut chain_code = root_chain_code;
227    let mut parent_fingerprint: [u8; 4] = [0u8; 4];
228
229    let path = chain_path.path();
230
231    let depth = path.len();
232
233    let final_child_num = if depth == 0 {
234        &ChildIndex::Normal(0)
235    } else {
236        &path[depth - 1]
237    };
238
239    for child_num in path {
240        parent_fingerprint = get_finger_print(&pubkey);
241        let (_, child_pubkey, child_chain_code) =
242            derive_child_pubkey(&pubkey, chain_code, child_num)?;
243        pubkey = child_pubkey;
244        chain_code = child_chain_code;
245    }
246
247    Ok(XPubKey {
248        prefix,
249        depth: depth as u8,
250        parent_fingerprint,
251        child_number: final_child_num.to_u32(),
252        chain_code,
253        pubkey,
254    })
255}
