13//! Another [Base58][] codec implementation.
14//!
15//! Compared to [`base58`][] this is significantly faster at decoding (about
16//! 2.4x as fast when decoding 32 bytes), almost the same speed for encoding
17//! (about 3% slower when encoding 32 bytes) and doesn't have the 128 byte
18//! limitation.
19//!
20//! Compared to [`rust-base58`][] this is massively faster (over ten times as
21//! fast when decoding 32 bytes, almost 40 times as fast when encoding 32
22//! bytes) and has no external dependencies.
23//!
24//! Compared to both this supports a configurable alphabet and user provided
25//! buffers for zero-allocation {en,de}coding.
26//!
27//! [Base58]: https://en.wikipedia.org/wiki/Base58
28//! [`base58`]: https://github.com/debris/base58
29//! [`rust-base58`]: https://github.com/nham/rust-base58
30//!
31//! # Features
32//!
33//!  Feature | Activation         | Effect
34//! ---------|--------------------|--------
35//!  `std`   | **on**-by-default  | Implement [`Error`](std::error::Error) for error types
36//!  `alloc` | implied by `std`   | Support encoding/decoding to [`Vec`](alloc::vec::Vec) and [`String`](alloc::string::String) as appropriate
37//!  `check` | **off**-by-default | Integrated support for [Base58Check][]
38//!
39//! [Base58Check]: https://en.bitcoin.it/wiki/Base58Check_encoding
40//!
41//! # Examples
42//!
43//! ## Basic example
44//!
45//! ```rust
46//! let decoded = bs58::decode("he11owor1d").into_vec()?;
47//! let encoded = bs58::encode(decoded).into_string();
48//! assert_eq!("he11owor1d", encoded);
49//! # Ok::<(), bs58::decode::Error>(())
50//! ```
51//!
52//! ## Changing the alphabet
53//!
54//! ```rust
55//! let decoded = bs58::decode("he11owor1d")
56//!     .with_alphabet(bs58::Alphabet::RIPPLE)
57//!     .into_vec()?;
58//! let encoded = bs58::encode(decoded)
59//!     .with_alphabet(bs58::Alphabet::FLICKR)
60//!     .into_string();
61//! assert_eq!("4DSSNaN1SC", encoded);
62//! # Ok::<(), bs58::decode::Error>(())
63//! ```
64//!
65//! ## Decoding into an existing buffer
66//!
67//! ```rust
68//! let (mut decoded, mut encoded) = ([0xFF; 8], String::with_capacity(10));
69//! bs58::decode("he11owor1d").into(&mut decoded)?;
70//! bs58::encode(decoded).into(&mut encoded)?;
71//! assert_eq!("he11owor1d", encoded);
72//! # Ok::<(), Box<dyn std::error::Error>>(())
73//! ```

98/// Setup decoder for the given string using the [default alphabet][Alphabet::DEFAULT].
99///
100/// # Examples
101///
102/// ## Basic example
103///
104/// ```rust
105/// assert_eq!(
106///     vec![0x04, 0x30, 0x5e, 0x2b, 0x24, 0x73, 0xf0, 0x58],
107///     bs58::decode("he11owor1d").into_vec()?);
108/// # Ok::<(), bs58::decode::Error>(())
109/// ```
110///
111/// ## Changing the alphabet
112///
113/// ```rust
114/// assert_eq!(
115///     vec![0x60, 0x65, 0xe7, 0x9b, 0xba, 0x2f, 0x78],
116///     bs58::decode("he11owor1d")
117///         .with_alphabet(bs58::Alphabet::RIPPLE)
118///         .into_vec()?);
119/// # Ok::<(), bs58::decode::Error>(())
120/// ```
121///
122/// ## Decoding into an existing buffer
123///
124/// ```rust
125/// let mut output = [0xFF; 10];
126/// assert_eq!(8, bs58::decode("he11owor1d").into(&mut output)?);
127/// assert_eq!(
128///     [0x04, 0x30, 0x5e, 0x2b, 0x24, 0x73, 0xf0, 0x58, 0xFF, 0xFF],
129///     output);
130/// # Ok::<(), bs58::decode::Error>(())
131/// ```
132///
133/// ## Errors
134///
135/// ### Invalid Character
136///
137/// ```rust
138/// assert_eq!(
139///     bs58::decode::Error::InvalidCharacter { character: 'l', index: 2 },
140///     bs58::decode("hello world").into_vec().unwrap_err());
141/// ```
142///
143/// ### Non-ASCII Character
144///
145/// ```rust
146/// assert_eq!(
147///     bs58::decode::Error::NonAsciiCharacter { index: 5 },
148///     bs58::decode("he11o🇳🇿").into_vec().unwrap_err());
149/// ```
150///
151/// ### Too Small Buffer
152///
153/// This error can only occur when reading into a provided buffer, when using
154/// [`into_vec()`][decode::DecodeBuilder::into_vec] a vector large enough is guaranteed to be
155/// used.
156///
157/// ```rust
158/// let mut output = [0; 7];
159/// assert_eq!(
160///     bs58::decode::Error::BufferTooSmall,
161///     bs58::decode("he11owor1d").into(&mut output).unwrap_err());
162/// ```
163pub fn decode<I: AsRef<[u8]>>(input: I) -> decode::DecodeBuilder<'static, I> {
164    decode::DecodeBuilder::from_input(input)
165}

167/// Setup encoder for the given bytes using the [default alphabet][Alphabet::DEFAULT].
168///
169/// # Examples
170///
171/// ## Basic example
172///
173/// ```rust
174/// let input = [0x04, 0x30, 0x5e, 0x2b, 0x24, 0x73, 0xf0, 0x58];
175/// assert_eq!("he11owor1d", bs58::encode(input).into_string());
176/// ```
177///
178/// ## Changing the alphabet
179///
180/// ```rust
181/// let input = [0x60, 0x65, 0xe7, 0x9b, 0xba, 0x2f, 0x78];
182/// assert_eq!(
183///     "he11owor1d",
184///     bs58::encode(input)
185///         .with_alphabet(bs58::Alphabet::RIPPLE)
186///         .into_string());
187/// ```
188///
189/// ## Encoding into an existing string
190///
191/// ```rust
192/// let input = [0x04, 0x30, 0x5e, 0x2b, 0x24, 0x73, 0xf0, 0x58];
193/// let mut output = "goodbye world".to_owned();
194/// bs58::encode(input).into(&mut output)?;
195/// assert_eq!("he11owor1d", output);
196/// # Ok::<(), bs58::encode::Error>(())
197/// ```
198///
199/// ## Errors
200///
201/// ### Too Small Buffer
202///
203/// This error can only occur when reading into an unresizeable buffer.
204///
205/// ```rust
206/// let input = [0x04, 0x30, 0x5e, 0x2b, 0x24, 0x73, 0xf0, 0x58];
207/// let mut output = [0; 7];
208/// assert_eq!(
209///     bs58::encode::Error::BufferTooSmall,
210///     bs58::encode(input).into(&mut output[..]).unwrap_err());
211/// ```
212pub fn encode<I: AsRef<[u8]>>(input: I) -> encode::EncodeBuilder<'static, I> {
213    encode::EncodeBuilder::from_input(input)
