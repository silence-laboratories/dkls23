26/// Keyshare for refresh of a party.
27#[derive(Clone, Zeroize)]
28pub struct KeyshareForRefresh {
29    /// Rank of each party
30    pub rank_list: Vec<u8>,
31
32    /// Threshold value
33    pub threshold: u8,
34
35    /// Public key of the generated key.
36    pub public_key: ProjectivePoint,
37
38    // Root chain code (used to derive child public keys)
39    pub(crate) root_chain_code: [u8; 32],
40
41    /// set s_i to None if party_i lost their key_share
42    pub s_i: Option<Scalar>,
43
44    /// set s_i to None if party_i lost their key_share
45    pub x_i_list: Option<Vec<NonZeroScalar>>,
46
47    /// list of participants ids who lost their key_shares,
48    /// should be in range [0, n-1]
49    pub lost_keyshare_party_ids: Vec<u8>,
50
51    /// Part ID from key share
52    pub party_id: u8,
53}
54
55impl fmt::Debug for KeyshareForRefresh {
56    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
57        f.debug_struct("KeyshareForRefresh")
58            .field("rank_list", &self.rank_list)
59            .field("threshold", &self.threshold)
60            .field("public_key", &self.public_key)
61            .field("lost_keyshare_party_ids", &self.lost_keyshare_party_ids)
62            .finish()
63    }
64}

128    ///  Serialize KeyshareForRefresh to bytes
129    ///  Used to send KeyshareForRefresh to other parties, for key-import
130    pub fn to_bytes(&self) -> Vec<u8> {
131        let mut bytes = Vec::with_capacity(self.size());
132
133        bytes.push(self.party_id);
134
135        bytes.push(self.rank_list.len() as u8);
136        bytes.extend_from_slice(&self.rank_list);
137
138        bytes.push(self.threshold);
139
140        bytes.extend_from_slice(&self.public_key.to_affine().to_bytes());
141
142        bytes.extend_from_slice(&self.root_chain_code);
143
144        if let Some(s_i) = self.s_i {
145            bytes.push(1);
146            bytes.extend_from_slice(&s_i.to_bytes());
147        } else {
148            bytes.push(0);
149        }
150
151        if let Some(x_i_list) = &self.x_i_list {
152            bytes.push(x_i_list.len() as u8);
153            for x_i in x_i_list {
154                bytes.extend_from_slice(&x_i.to_bytes());
155            }
156        } else {
157            bytes.push(0);
158        }
159
160        bytes.push(self.lost_keyshare_party_ids.len() as u8);
161        bytes.extend_from_slice(&self.lost_keyshare_party_ids);
162
163        bytes
164    }
165
166    /// Deserialize KeyshareForRefresh from bytes
167    pub fn from_bytes(bytes: &[u8]) -> Option<Self> {
168        let offset = std::cell::Cell::new(0usize);
169
170        let read_data = |num_bytes: u8| {
171            let len = num_bytes as usize;
172            let off = offset.replace(offset.get() + len);
173            bytes.get(off..off + len)
174        };
175
176        let read_byte = || read_data(1).map(|b| b[0]);
177
178        let party_id = read_byte()?;
179
180        let rank_list_len = read_byte()?;
181        let rank_list = read_data(rank_list_len)?;
182
183        let threshold = read_byte()?;
184
185        let public_key =
186            read_data(33).and_then(|b| ProjectivePoint::from_bytes(b.into()).into_option())?;
187
188        let root_chain_code: [u8; 32] = read_data(32)?.try_into().ok()?;
189
190        let s_i = if read_byte()? == 1 {
191            let s_i_bytes: [u8; 32] = read_data(32)?.try_into().ok()?;
192            Scalar::from_repr(s_i_bytes.into()).into()
193        } else {
194            None
195        };
196
197        let x_i_list_len = read_byte()?;
198        let x_i_list = if x_i_list_len != 0 {
199            let mut x_i_list = Vec::with_capacity(x_i_list_len as usize);
200            for _ in 0..x_i_list_len {
201                let x_i_bytes: [u8; 32] = read_data(32)?.try_into().ok()?;
202                let x_i: NonZeroScalar = Option::from(NonZeroScalar::from_repr(x_i_bytes.into()))?;
203                x_i_list.push(x_i);
204            }
205            Some(x_i_list)
206        } else {
207            None
208        };
209
210        let lost_keyshare_party_ids_len = read_byte()?;
211        let lost_keyshare_party_ids = read_data(lost_keyshare_party_ids_len)?;
212
213        Some(Self {
214            rank_list: rank_list.to_vec(),
215            threshold,
216            public_key,
217            root_chain_code,
218            s_i,
219            x_i_list,
220            lost_keyshare_party_ids: lost_keyshare_party_ids.to_vec(),
221            party_id,
222        })
223    }
224
225    fn size(&self) -> usize {
226        let mut size = 1 + self.rank_list.len();
227        size += 1; // party_id
228        size += 1;
229        size += 33;
230        size += 32;
231        size += 1;
232        if self.s_i.is_some() {
233            size += 32;
234        }
235        if let Some(x_i_list) = &self.x_i_list {
236            size += 1;
237            size += x_i_list.len() * 32;
238        } else {
239            size += 1;
240        }
241        size += 1 + self.lost_keyshare_party_ids.len();
242        size
243    }
244}

246/// Execute Key Refresh protocol.
247pub async fn run<R, S>(
248    setup: S,
249    seed: Seed,
250    relay: R,
251    old_keyshare: KeyshareForRefresh,
252) -> Result<Keyshare, KeygenError>
253where
254    S: KeygenSetupMessage,
255    R: Relay,
256{
257    let abort_msg = create_abort_message(&setup);
258    let mut relay = FilteredMsgRelay::new(relay);
259
260    let my_party_id = old_keyshare.party_id;
261    let n = setup.total_participants();
262
263    let mut s_i_0 = Scalar::ZERO;
264    if old_keyshare.s_i.is_some() && old_keyshare.x_i_list.is_some() {
265        // calculate additive share s_i_0 of participant_i,
266        // \sum_{i=0}^{n-1} s_i_0 = private_key
267        let s_i = &old_keyshare.s_i.unwrap();
268        let rank_list = &old_keyshare.rank_list;
269        let x_i_list = &old_keyshare.x_i_list.unwrap();
270        let x_i = &x_i_list[my_party_id as usize];
271
272        let party_ids_with_keyshares = (0..n as u8)
273            .filter(|p| !old_keyshare.lost_keyshare_party_ids.contains(&{ *p }))
274            .collect::<Vec<_>>();
275
276        let all_ranks_zero = rank_list.iter().all(|r| r == &0u8);
277
278        let lambda = if all_ranks_zero {
279            get_lagrange_coeff(x_i, x_i_list, &party_ids_with_keyshares)
280        } else {
281            get_birkhoff_coefficients(rank_list, x_i_list, &party_ids_with_keyshares)
282                .get(&(my_party_id as usize))
283                .cloned()
284                .unwrap_or(Scalar::ZERO)
285        };
286
287        s_i_0 = lambda * s_i;
288    }
289
290    let key_refresh_data = KeyRefreshData {
291        s_i_0,
292        lost_keyshare_party_ids: old_keyshare.lost_keyshare_party_ids,
293        expected_public_key: old_keyshare.public_key,
294        root_chain_code: old_keyshare.root_chain_code,
295    };
296
297    let result: Result<Keyshare, KeygenError> =
298        run_inner(setup, seed, &mut relay, Some(&key_refresh_data)).await;
299
300    let new_keyshare = match result {
301        Ok(eph_keyshare) => eph_keyshare,
302
303        Err(KeygenError::AbortProtocol(p)) => return Err(KeygenError::AbortProtocol(p)),
304
305        Err(KeygenError::SendMessage) => return Err(KeygenError::SendMessage),
306
307        Err(err_message) => {
308            #[cfg(feature = "tracing")]
309            tracing::debug!("sending abort message");
310
311            relay.send(abort_msg).await?;
312
313            return Err(err_message);
314        }
315    };
316
317    Ok(new_keyshare)
318}

320/// Generate ValidatedSetup and seed for Key refresh
321#[cfg(any(test, feature = "test-support"))]
322pub fn setup_key_refresh(
323    t: u8,
324    n: u8,
325    n_i_list: Option<&[u8]>,
326    key_shares_for_refresh: Vec<KeyshareForRefresh>,
327) -> Vec<(
328    crate::setup::keygen::SetupMessage,
329    [u8; 32],
330    KeyshareForRefresh,
331)> {
332    super::utils::setup_keygen(None, t, n, n_i_list)
333        .into_iter()
334        .zip(key_shares_for_refresh)
335        .map(|((setup, seed), share)| (setup, seed, share))
336        .collect()
337}

496    #[test]
497    fn refresh_ser_de() {
498        let share = KeyshareForRefresh::new(
499            vec![0, 0, 0, 0],
500            2,
501            ProjectivePoint::GENERATOR * Scalar::ONE,
502            [0u8; 32],
503            Some(Scalar::ONE),
504            Some(vec![NonZeroScalar::new(Scalar::ONE).unwrap()]),
505            vec![0, 1],
506            0,
507        );
508
509        let bytes = share.to_bytes();
510        let share2 = KeyshareForRefresh::from_bytes(&bytes).unwrap();
511
512        assert_eq!(share.rank_list, share2.rank_list);
513        assert_eq!(share.threshold, share2.threshold);
514        assert_eq!(share.public_key, share2.public_key);
515        assert_eq!(share.root_chain_code, share2.root_chain_code);
516        assert_eq!(share.s_i, share2.s_i);
517        assert_eq!(share.x_i_list.is_some(), share2.x_i_list.is_some());
518        let x_i_list = share.x_i_list.unwrap();
519        let x_i_list2 = share2.x_i_list.unwrap();
520
521        assert_eq!(x_i_list.len(), x_i_list2.len());
522
523        for (x_i, x_i2) in x_i_list.iter().zip(x_i_list2.iter()) {
524            assert_eq!(x_i.to_bytes(), x_i2.to_bytes());
525        }
526
527        assert_eq!(
528            share.lost_keyshare_party_ids,
529            share2.lost_keyshare_party_ids
530        );
531    }
