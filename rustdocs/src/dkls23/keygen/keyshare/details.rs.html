4//! Internal details of the key share implementation
5//!
6//! This module contains the core data structures that make up a key share in the
7//! distributed key generation protocol. These structures are designed to be
8//! memory-safe and efficiently serializable.
9//!
10//! # Safety
11//!
12//! The structures in this module use `bytemuck` traits to ensure safe memory
13//! operations and alignment requirements. All structures are marked with
14//! `#[repr(C)]` to guarantee a stable memory layout.
15
16use bytemuck::{AnyBitPattern, NoUninit};
17
18use sl_oblivious::soft_spoken::{ReceiverOTSeed, SenderOTSeed};
19
20use crate::proto::*;
21
22/// Core information about a key share
23///
24/// This structure contains all the essential information about a key share,
25/// including protocol parameters, cryptographic material, and metadata.
26///
27/// # Memory Layout
28///
29/// The structure is marked with `#[repr(C)]` to ensure a stable memory layout
30/// and uses `AnyBitPattern` and `NoUninit` for safe memory operations.
31///
32/// # Fields
33///
34/// * `magic` - A marker to identify valid key shares
35/// * `extra` - Size of additional data associated with the key share
36/// * `total_parties` - Total number of participants in the protocol
37/// * `threshold` - The threshold value for key reconstruction
38/// * `party_id` - The ID of the party holding this share
39/// * `final_session_id` - The final session identifier
40/// * `root_chain_code` - The root chain code for key derivation
41/// * `public_key` - The public key of the generated key
42/// * `key_id` - A unique identifier for the key
43/// * `s_i` - The secret share value
44#[derive(Clone, Copy, AnyBitPattern, NoUninit)]
45#[repr(C)]
46pub struct KeyshareInfo {
47    /// A marker to identify valid key shares
48    pub magic: [u8; 4],
49
50    /// Size in bytes of extra data associated with Keyshare
51    pub(crate) extra: [u8; 4],
52
53    /// Total number of parties in the protocol
54    pub total_parties: u8,
55
56    /// Threshold value for key reconstruction
57    pub threshold: u8,
58
59    /// Party ID of the sender
60    pub party_id: u8,
61
62    /// Final session ID from the protocol
63    pub final_session_id: [u8; 32],
64
65    /// Root chain code (used to derive child public keys)
66    pub root_chain_code: [u8; 32],
67
68    /// Public key of the generated key
69    pub public_key: PointBytes,
70
71    /// Key ID for identification
72    pub key_id: [u8; 32],
73
74    /// Secret share value
75    pub(crate) s_i: ScalarBytes,
76}
77
78// Ensure proper memory alignment
79const _: () = assert!(core::mem::align_of::<KeyshareInfo>() == 1);

81/// Information about another party in the protocol
82///
83/// This structure contains the oblivious transfer seeds used for communication
84/// with another party in the protocol.
85///
86/// # Memory Layout
87///
88/// The structure is marked with `#[repr(C)]` to ensure a stable memory layout
89/// and uses `AnyBitPattern` and `NoUninit` for safe memory operations.
90///
91/// # Fields
92///
93/// * `send_ot_seed` - Seed for sending oblivious transfer messages
94/// * `recv_ot_seed` - Seed for receiving oblivious transfer messages
95#[derive(Clone, Copy, AnyBitPattern, NoUninit)]
96#[repr(C)]
97pub(crate) struct OtherParty {
98    /// Seed for sending oblivious transfer messages
99    pub(crate) send_ot_seed: SenderOTSeed,
100
101    /// Seed for receiving oblivious transfer messages
102    pub(crate) recv_ot_seed: ReceiverOTSeed,
103}
104
105// Ensure proper memory alignment
106const _: () = assert!(core::mem::align_of::<OtherParty>() == 1);

108/// Information about each party in the protocol
109///
110/// This structure contains the necessary information about each party's
111/// contribution to the key share.
112///
113/// # Memory Layout
114///
115/// The structure is marked with `#[repr(C)]` to ensure a stable memory layout
116/// and uses `AnyBitPattern` and `NoUninit` for safe memory operations.
117///
118/// # Fields
119///
120/// * `rank` - The rank of the party
121/// * `big_s` - The party's contribution to the public key
122/// * `x_i` - The x-coordinate for the party's share (must be non-zero)
123/// * `zeta_seed` - A random seed for additional security
124#[derive(Clone, Copy, AnyBitPattern, NoUninit)]
125#[repr(C)]
126pub(crate) struct EachParty {
127    /// The rank of the party
128    pub(crate) rank: u8,
129
130    /// The party's contribution to the public key
131    pub(crate) big_s: PointBytes,
132
133    /// The x-coordinate for the party's share (must be non-zero)
134    pub(crate) x_i: ScalarBytes,
135
136    /// A random seed for additional security
137    pub(crate) zeta_seed: [u8; 32],
138}
139
140// Ensure proper memory alignment
141const _: () = assert!(core::mem::align_of::<EachParty>() == 1);
