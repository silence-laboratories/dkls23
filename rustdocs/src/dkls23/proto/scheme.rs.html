29/// Represents an encryption scheme interface for in-place AEAD and
30/// managing key exchange.
31///
32/// A type implementing EncryptionScheme encapsulates two
33/// functionalities:
34///
35/// - An AEAD algorithm such as ChaCha20 or AES-GCM and its
36///   implementation details, such as nonce generation.
37///
38/// - The derivation of encryption keys for one or more pairs of parties.
39///   The concrete type of public/private key pair and key exchange is
40///   an implementation detail.
41///
42pub trait EncryptionScheme: Send {
43    /// Return external representation of own public key
44    fn public_key(&self) -> &[u8];
45
46    /// Sets or updates the public key for a specified receiver index.
47    ///
48    /// This function is used to associate a public key with a designated
49    /// receiver identified by an index. It facilitates secure communication
50    /// setup by ensuring that messages can be encrypted such that only the
51    /// receiver with the corresponding private key can decrypt them.
52    ///
53    /// # Parameters
54    ///
55    /// - `receiver_index`: An integer value that uniquely identifies the
56    ///   receiver within the system. This index is used to specify which
57    ///   receiver the public key is associated with.
58    ///
59    /// - `public_key`: A byte slice representing the public key of the receiver.
60    ///   This key is used in cryptographic operations to ensure that only
61    ///   the intended receiver can decrypt messages encrypted with this key.
62    ///
63    ///
64    /// # Errors
65    ///
66    /// - `PublicKeyError`: An error is returned if the public key cannot be
67    ///   set due to invalid formatting, an invalid receiver index, or any
68    ///   other issue encountered in the operation. The precise error variant
69    ///   provides further details on the nature of the failure.
70    ///
71    fn receiver_public_key(
72        &mut self,
73        receiver_index: usize,
74        public_key: &[u8],
75    ) -> Result<(), PublicKeyError>;
76
77    /// Encrypts the provided data buffer using associated data and
78    /// manages an associated tail segment.
79    ///
80    /// # Parameters
81    ///
82    /// - `associated_data`: A byte slice containing additional
83    ///    authenticated data (AAD) that will be used to ensure the
84    ///    integrity and authenticity of the encrypted data. This data is
85    ///    not encrypted but is included in the integrity check.
86    ///
87    /// - `buffer`: A mutable byte slice containing the plaintext data
88    ///   that will be encrypted in place. Upon successful encryption,
89    ///    this buffer will contain the ciphertext data.
90    ///
91    /// - `tail`: A mutable byte slice representing the trailing
92    ///   segment of the data buffer that may be used to store for
93    ///   trailing portion of the data that should be considered during
94    ///   decryption.
95    ///
96    /// - `receive`: An index or identifier associated with the
97    ///   receiver of the message. This may be used for deriving
98    ///   encryption keys.
99    ///
100    /// # Errors
101    ///
102    ///   `EncryptionError` if issues arise such as missing keys,
103    ///   incorrect buffer lengths, or any other problems during the
104    ///   encryption process. The error provides specific details about
105    ///   the nature of the failure.
106    ///
107    fn encrypt(
108        &mut self,
109        associated_data: &[u8],
110        buffer: &mut [u8],
111        tail: &mut [u8],
112        receive: usize,
113    ) -> Result<(), EncryptionError>;
114
115    /// Decrypts the provided data buffer using associated data and a
116    /// tail segment.
117    ///
118    /// # Parameters
119    ///
120    /// - `associated_data`: A byte slice containing additional
121    ///   authenticated data (AAD) that will be used along with the
122    ///   buffer to ensure the integrity and authenticity of the
123    ///   decryption process. This data is not encrypted but plays a
124    ///   role in verifying the encrypted data.
125    ///
126    /// - `buffer`: A mutable byte slice holding the encrypted data
127    ///   that will be decrypted in place. Upon successful decryption,
128    ///   this buffer will contain the plaintext data.
129    ///
130    /// - `tail`: A byte slice representing the trailing portion of
131    ///   the data that should be considered during decryption. This is
132    ///   typically used nonce and/or authentication tag.
133    ///
134    /// - `sender`: An index or identifier representing the sender of
135    ///   the message. This might be used to retrieve or derive
136    ///   encryption keys.
137    ///
138    /// # Errors
139    ///
140    /// - `EncryptionError`: This function may return an
141    ///   `EncryptionError` in several situations such as when the
142    ///   decryption key is not found, when the input data is tampered
143    ///   with or if the cryptographic verification of the AAD fails.
144    ///
145    fn decrypt(
146        &self,
147        associated_data: &[u8],
148        buffer: &mut [u8],
149        tail: &[u8],
150        sender: usize,
151    ) -> Result<(), EncryptionError>;
152
153    /// Return size of trailing segment. See method `encrypt()` and `decrypt()`.
154    fn overhead(&self) -> usize;
155}

157/// Counter to create a unuque nonce.
158#[derive(Default)]
159pub struct NonceCounter(u32);
160
161impl NonceCounter {
162    /// New counter initialized by 0.
163    fn new() -> Self {
164        Self(0)
165    }
166
167    /// Increment counter.
168    fn next_nonce<S: AeadCore>(&mut self) -> Nonce<S> {
169        // In our design, we use 3-5 nonces per unique symmetric key.
170        // If the u32 counter overflows, then calling it is definitely
171        // a misuse of the counter, and it's safer to crash than reuse
172        // a nonce.
173        self.0 = self.0.checked_add(1).expect("nonce overflow");
174
175        let mut nonce = Nonce::<S>::default();
176        nonce[..4].copy_from_slice(&self.0.to_le_bytes());
177
178        nonce
179    }
180}

182/// The implementation of EncryptionScheme that uses x25519 for key
183/// exchange and any implementation of `AeadInPlace`.
184pub struct AeadX25519<S> {
185    secret: ReusableSecret,
186    public_key: PublicKey,
187    counter: NonceCounter,
188    pk: Pairs<(SharedKey, PublicKey), usize>,
189    marker: PhantomData<S>,
190}
191
192impl<S> AeadX25519<S> {
193    /// Generate a new [`AeadX25519`] with the supplied RNG.
194    pub fn new(rng: &mut impl CryptoRngCore) -> Self {
195        let secret = ReusableSecret::random_from_rng(rng);
196        let public_key = PublicKey::from(&secret);
197
198        Self {
199            secret,
200            public_key,
201            counter: NonceCounter::new(),
202            pk: Pairs::new(),
203            marker: PhantomData,
204        }
205    }
206
207    /// Create a new [`AeadX25519`] from provided `ReusableSecret`.
208    pub fn from_secret(secret: ReusableSecret) -> Self {
209        let public_key = PublicKey::from(&secret);
210
211        Self {
212            secret,
213            public_key,
214            counter: NonceCounter::new(),
215            pk: Pairs::new(),
216            marker: PhantomData,
217        }
218    }
219}
220
221impl<S> EncryptionScheme for AeadX25519<S>
222where
223    S: AeadInPlace + KeyInit + Send,
224{
225    fn overhead(&self) -> usize {
226        S::TagSize::USIZE + S::NonceSize::USIZE
227    }
228
229    fn encrypt(
230        &mut self,
231        associated_data: &[u8],
232        buffer: &mut [u8],
233        tail: &mut [u8],
234        receiver: usize,
235    ) -> Result<(), EncryptionError> {
236        if tail.len() != self.overhead() {
237            return Err(EncryptionError);
238        }
239
240        let (key, public_key) =
241            self.pk.find_pair_or_err(receiver, EncryptionError)?;
242
243        let key = Zeroizing::new(
244            Sha256::new_with_prefix(public_key)
245                .chain_update(key)
246                .finalize(),
247        );
248
249        let key = Key::<S>::from_slice(key.as_slice());
250
251        let nonce = self.counter.next_nonce::<S>();
252        let tag = S::new(key)
253            .encrypt_in_place_detached(&nonce, associated_data, buffer)
254            .map_err(|_| EncryptionError)?;
255
256        tail[..S::TagSize::USIZE].copy_from_slice(&tag);
257        tail[S::TagSize::USIZE..].copy_from_slice(&nonce);
258
259        Ok(())
260    }
261
262    fn decrypt(
263        &self,
264        associated_data: &[u8],
265        buffer: &mut [u8],
266        tail: &[u8],
267        sender: usize,
268    ) -> Result<(), EncryptionError> {
269        if tail.len() != self.overhead() {
270            return Err(EncryptionError);
271        }
272
273        let (key, public_key) =
274            self.pk.find_pair_or_err(sender, EncryptionError)?;
275
276        let key = Zeroizing::new(
277            Sha256::new_with_prefix(self.public_key)
278                .chain_update(key)
279                .finalize(),
280        );
281
282        let key = Key::<S>::from_slice(key.as_slice());
283
284        let nonce = Nonce::<S>::from_slice(&tail[S::TagSize::USIZE..]);
285        let tag = Tag::<S>::from_slice(&tail[..S::TagSize::USIZE]);
286
287        S::new(key)
288            .decrypt_in_place_detached(nonce, associated_data, buffer, tag)
289            .map_err(|_| EncryptionError)?;
290
291        Ok(())
292    }
293
294    fn public_key(&self) -> &[u8] {
295        self.public_key.as_bytes()
296    }
297
298    fn receiver_public_key(
299        &mut self,
300        receiver_index: usize,
301        pk: &[u8],
302    ) -> Result<(), PublicKeyError> {
303        let pk: [u8; 32] = pk.try_into().map_err(|_| PublicKeyError)?;
304        let pk = PublicKey::from(pk);
305
306        let shared_secret = self.secret.diffie_hellman(&pk);
307
308        if !shared_secret.was_contributory() {
309            return Err(PublicKeyError);
310        }
311
312        let shared_key = Zeroizing::new(hchacha::<U10>(
313            GenericArray::from_slice(shared_secret.as_bytes()),
314            &GenericArray::default(),
315        ));
316
317        self.pk.push(receiver_index, (shared_key, pk));
318
319        Ok(())
320    }
321}
