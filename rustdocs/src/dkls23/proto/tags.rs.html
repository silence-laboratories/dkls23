23/// Relay Errors
24pub enum Error {
25    /// Abort
26    Abort(usize),
27    /// Recv
28    Recv,
29    /// Send
30    Send,
31    /// InvalidMessage
32    InvalidMessage,
33}

35/// custom message relay
36pub struct FilteredMsgRelay<R> {
37    relay: R,
38    in_buf: Vec<(Vec<u8>, usize, MessageTag)>,
39    expected: HashMap<MsgId, (usize, MessageTag)>,
40}
41
42impl<R: Relay> FilteredMsgRelay<R> {
43    /// Construct a FilteredMsgRelay by wrapping up a Relay object
44    pub fn new(relay: R) -> Self {
45        Self {
46            relay,
47            expected: HashMap::new(),
48            in_buf: vec![],
49        }
50    }
51
52    /// Return underlying relay object
53    pub fn into_inner(self) -> R {
54        self.relay
55    }
56
57    /// Mark message with ID as expected and associate pair (party-id,
58    /// tag) with it.
59    pub async fn expect_message(
60        &mut self,
61        id: MsgId,
62        tag: MessageTag,
63        party_id: usize,
64        ttl: u32,
65    ) -> Result<(), MessageSendError> {
66        self.relay.ask(&id, ttl).await?;
67        self.expected.insert(id, (party_id, tag));
68
69        Ok(())
70    }
71
72    fn put_back(&mut self, msg: &[u8], tag: MessageTag, party_id: usize) {
73        self.expected
74            .insert(msg.try_into().unwrap(), (party_id, tag));
75    }
76
77    /// Receive an expected message with given tag, and return a
78    /// party-id associated with it.
79    pub async fn recv(
80        &mut self,
81        tag: MessageTag,
82    ) -> Result<(Vec<u8>, usize, bool), Error> {
83        // flush output message messages.
84        self.relay.flush().await.map_err(|_| Error::Recv)?;
85
86        if let Some(idx) = self.in_buf.iter().position(|ent| ent.2 == tag) {
87            let (msg, p, _) = self.in_buf.swap_remove(idx);
88            return Ok((msg, p, false));
89        }
90
91        loop {
92            let msg = self.relay.next().await.ok_or(Error::Recv)?;
93
94            if let Ok(id) = <&MsgId>::try_from(msg.as_slice()) {
95                if let Some(&(p, t)) = self.expected.get(id) {
96                    self.expected.remove(id);
97                    match t {
98                        ABORT_MESSAGE_TAG => {
99                            return Ok((msg, p, true));
100                        }
101
102                        _ if t == tag => {
103                            return Ok((msg, p, false));
104                        }
105
106                        _ => {
107                            // some expected but not required right
108                            // now message.
109                            self.in_buf.push((msg, p, t));
110                        }
111                    }
112                }
113            }
114        }
115    }

117    /// Add expected messages and Ask underlying message relay to
118    /// receive them.
119    pub async fn ask_messages<P: ProtocolParticipant>(
120        &mut self,
121        setup: &P,
122        tag: MessageTag,
123        p2p: bool,
124    ) -> Result<usize, MessageSendError> {
125        self.ask_messages_from_iter(
126            setup,
127            tag,
128            setup.all_other_parties(),
129            p2p,
130        )
131        .await
132    }
133
134    /// Ask set of messages with a given `tag` from a set of `parties`.
135    ///
136    /// Filter out own `party_index` from `parties`.
137    ///
138    /// Returns number of messages with the same tag.
139    ///
140    pub async fn ask_messages_from_iter<P, I>(
141        &mut self,
142        setup: &P,
143        tag: MessageTag,
144        from_parties: I,
145        p2p: bool,
146    ) -> Result<usize, MessageSendError>
147    where
148        P: ProtocolParticipant,
149        I: IntoIterator<Item = usize>,
150    {
151        let my_party_index = setup.participant_index();
152        let receiver = p2p.then_some(my_party_index);
153        let mut count = 0;
154        for sender_index in from_parties.into_iter() {
155            if sender_index == my_party_index {
156                continue;
157            }
158
159            count += 1;
160            self.expect_message(
161                setup.msg_id_from(sender_index, receiver, tag),
162                tag,
163                sender_index,
164                setup.message_ttl().as_secs() as _,
165            )
166            .await?;
167        }
168
169        Ok(count)
170    }
171
172    /// The same as `ask_messages_from_iter()` by accepts slice of indices
173    pub async fn ask_messages_from_slice<'a, P, I>(
174        &mut self,
175        setup: &P,
176        tag: MessageTag,
177        from_parties: I,
178        p2p: bool,
179    ) -> Result<usize, MessageSendError>
180    where
181        P: ProtocolParticipant,
182        I: IntoIterator<Item = &'a usize>,
183    {
184        self.ask_messages_from_iter(
185            setup,
186            tag,
187            from_parties.into_iter().copied(),
188            p2p,
189        )
190        .await
191    }

195        Round::new(count, tag, self)
196    }

213/// Structure to receive a round of messages
214pub struct Round<'a, R> {
215    tag: MessageTag,
216    count: usize,
217    pub(crate) relay: &'a mut FilteredMsgRelay<R>,
218}
219
220impl<'a, R: Relay> Round<'a, R> {
221    /// Create a new round with a given number of messages to receive.
222    pub fn new(
223        count: usize,
224        tag: MessageTag,
225        relay: &'a mut FilteredMsgRelay<R>,
226    ) -> Self {
227        Self { count, tag, relay }
228    }
229
230    /// Receive next message in the round.
231    /// On success returns Ok(Some(message, party_index, is_abort_flag)).
232    /// At the end of the round it returns Ok(None).
233    ///
234    pub async fn recv(
235        &mut self,
236    ) -> Result<Option<(Vec<u8>, usize, bool)>, Error> {
237        Ok(if self.count > 0 {
238            let msg = self.relay.recv(self.tag).await;
239            #[cfg(feature = "tracing")]
240            if msg.is_err() {
241                for (id, (p, t)) in &self.relay.expected {
242                    if t == &self.tag {
243                        tracing::debug!("waiting for {:X} {} {:?}", id, p, t);
244                    }
245                }
246            }
247            let msg = msg?;
248            self.count -= 1;
249            Some(msg)
250        } else {
251            None
252        })
253    }
254
255    /// It is possible to receive a invalid message with a correct ID.
256    /// In this case, it have to put the message id back into
257    /// relay.expected table and increment a counter of waiting
258    /// messages in the round.
259    pub fn put_back(&mut self, msg: &[u8], tag: MessageTag, party_id: usize) {
260        self.relay.put_back(msg, tag, party_id);
261        self.count += 1;
262
263        // TODO Should we ASK it again?
264    }

266    /// Receiver all messages in the round, verify, decode and pass to
267    /// given handler.
268    pub async fn of_signed_messages<T, F, S, E>(
269        mut self,
270        setup: &S,
271        abort_err: impl Fn(usize) -> E,
272        mut handler: F,
273    ) -> Result<(), E>
274    where
275        T: AnyBitPattern + NoUninit,
276        S: ProtocolParticipant,
277        F: FnMut(&T, usize) -> Result<(), E>,
278        E: From<Error>,
279    {
280        while let Some((msg, party_idx, is_abort)) = self.recv().await? {
281            if is_abort {
282                check_abort(setup, &msg, party_idx, &abort_err)?;
283                self.put_back(&msg, ABORT_MESSAGE_TAG, party_idx);
284                continue;
285            }
286
287            let vk = setup.verifier(party_idx);
288            let msg: &T = match SignedMessage::verify(&msg, vk) {
289                Some(refs) => refs,
290                _ => {
291                    self.put_back(&msg, self.tag, party_idx);
292                    continue;
293                }
294            };
295
296            handler(msg, party_idx)?;
297        }
298
299        Ok(())
300    }
301
302    /// Receiver all messages in the round, decrypt, decode and pass
303    /// to given handler.
304    pub async fn of_encrypted_messages<T, F, P, E>(
305        mut self,
306        setup: &P,
307        scheme: &mut dyn EncryptionScheme,
308        trailer: usize,
309        err: impl Fn(usize) -> E,
310        mut handler: F,
311    ) -> Result<(), E>
312    where
313        T: AnyBitPattern + NoUninit,
314        P: ProtocolParticipant,
315        F: FnMut(
316            &T,
317            usize,
318            &[u8],
319            &mut dyn EncryptionScheme,
320        ) -> Result<Option<Vec<u8>>, E>,
321        E: From<Error>,
322    {
323        while let Some((msg, party_index, is_abort)) = self.recv().await? {
324            if is_abort {
325                check_abort(setup, &msg, party_index, &err)?;
326                self.put_back(&msg, ABORT_MESSAGE_TAG, party_index);
327                continue;
328            }
329
330            let mut msg = Zeroizing::new(msg);
331
332            let (msg, trailer) = match EncryptedMessage::<T>::decrypt(
333                &mut msg,
334                trailer,
335                scheme,
336                party_index,
337            ) {
338                Some(refs) => refs,
339                _ => {
340                    self.put_back(&msg, self.tag, party_index);
341                    continue;
342                }
343            };
344
345            if let Some(replay) = handler(msg, party_index, trailer, scheme)?
346            {
347                self.relay.send(replay).await.map_err(|_| Error::Send)?;
348            }
349        }
350
351        Ok(())
352    }

354    /// Broadcast 4 values and collect 4 values from others in the round
355    pub async fn broadcast_4<P, T1, T2, T3, T4>(
356        self,
357        setup: &P,
358        msg: (T1, T2, T3, T4),
359    ) -> Result<
360        (
361            Pairs<T1, usize>,
362            Pairs<T2, usize>,
363            Pairs<T3, usize>,
364            Pairs<T4, usize>,
365        ),
366        Error,
367    >
368    where
369        P: ProtocolParticipant,
370        T1: Wrap,
371        T2: Wrap,
372        T3: Wrap,
373        T4: Wrap,
374    {
375        #[cfg(feature = "tracing")]
376        tracing::debug!("enter broadcast {:?}", self.tag);
377
378        let my_party_id = setup.participant_index();
379
380        let sizes = [
381            msg.0.external_size(),
382            msg.1.external_size(),
383            msg.2.external_size(),
384            msg.3.external_size(),
385        ];
386        let trailer: usize = sizes.iter().sum();
387
388        let buffer = {
389            // Do not hold SignedMessage across an await point to avoid
390            // forcing ProtocolParticipant::MessageSignature to be Send
391            // in case if the future returned by run() have to be Send.
392            let mut buffer = SignedMessage::<(), _>::new(
393                &setup.msg_id(None, self.tag),
394                setup.message_ttl().as_secs() as _,
395                0,
396                trailer,
397            );
398
399            let (_, mut out) = buffer.payload();
400
401            out = msg.0.encode(out);
402            out = msg.1.encode(out);
403            out = msg.2.encode(out);
404            msg.3.encode(out);
405
406            buffer.sign(setup.signer())
407        };
408
409        self.relay.send(buffer).await.map_err(|_| Error::Send)?;
410
411        let (mut p0, mut p1, mut p2, mut p3) =
412            self.recv_broadcast_4(setup, &sizes).await?;
413
414        p0.push(my_party_id, msg.0);
415        p1.push(my_party_id, msg.1);
416        p2.push(my_party_id, msg.2);
417        p3.push(my_party_id, msg.3);
418
419        Ok((p0, p1, p2, p3))
420    }

422    /// Receive broadcasted 4 values from all parties in the round and
423    /// collect them into 4 Pairs vectors.
424    pub async fn recv_broadcast_4<P, T1, T2, T3, T4>(
425        mut self,
426        setup: &P,
427        sizes: &[usize; 4],
428    ) -> Result<
429        (
430            Pairs<T1, usize>,
431            Pairs<T2, usize>,
432            Pairs<T3, usize>,
433            Pairs<T4, usize>,
434        ),
435        Error,
436    >
437    where
438        P: ProtocolParticipant,
439        T1: Wrap,
440        T2: Wrap,
441        T3: Wrap,
442        T4: Wrap,
443    {
444        let trailer: usize = sizes.iter().sum();
445
446        let mut p0 = Pairs::new();
447        let mut p1 = Pairs::new();
448        let mut p2 = Pairs::new();
449        let mut p3 = Pairs::new();
450
451        while let Some((msg, party_id, is_abort)) = self.recv().await? {
452            if is_abort {
453                check_abort(setup, &msg, party_id, Error::Abort)?;
454                self.put_back(&msg, ABORT_MESSAGE_TAG, party_id);
455                continue;
456            }
457
458            let buf = match SignedMessage::<(), _>::verify_with_trailer(
459                &msg,
460                trailer,
461                setup.verifier(party_id),
462            ) {
463                Some((_, msg)) => msg,
464                None => {
465                    // We got message with a right ID but with broken signature.
466                    self.put_back(&msg, self.tag, party_id);
467                    continue;
468                }
469            };
470
471            let (buf, v1) =
472                T1::decode(buf, sizes[0]).ok_or(Error::InvalidMessage)?;
473            let (buf, v2) =
474                T2::decode(buf, sizes[1]).ok_or(Error::InvalidMessage)?;
475            let (buf, v3) =
476                T3::decode(buf, sizes[2]).ok_or(Error::InvalidMessage)?;
477            let (_bu, v4) =
478                T4::decode(buf, sizes[3]).ok_or(Error::InvalidMessage)?;
479
480            p0.push(party_id, v1);
481            p1.push(party_id, v2);
482            p2.push(party_id, v3);
483            p3.push(party_id, v4);
484        }
485
486        Ok((p0, p1, p2, p3))
487    }
