4use crate::proto::tags::Error;
5use sl_mpc_mate::coord::MessageSendError;
6
7/// Distributed key generation errors
8#[derive(thiserror::Error, Debug)]
9#[allow(missing_docs)]
10pub enum SignError {
11    /// Invalid commitment
12    #[error("Invalid commitment")]
13    InvalidCommitment,
14
15    /// Invalid digest
16    #[error("Invalid digest")]
17    InvalidDigest,
18
19    /// Invalid final_session_id
20    #[error("Invalid final_session_id")]
21    InvalidFinalSessionID,
22
23    #[error("Failed check: {0}")]
24    /// Failed check
25    FailedCheck(&'static str),
26
27    /// k256 error
28    #[error("k256 error")]
29    K256Error,
30
31    /// Invalid PreSignature
32    #[error("invalid pre signature")]
33    InvalidPreSign,
34
35    /// Invalid message format
36    #[error("invalid message format")]
37    InvalidMessage,
38
39    #[error("Missing message")]
40    MissingMessage,
41
42    /// We can't a send message
43    #[error("Send message")]
44    SendMessage,
45
46    /// Some party decided to not participate in the protocol.
47    #[error("Abort protocol by party {0}")]
48    AbortProtocol(usize),
49
50    /// Abort the protocol and ban the party
51    #[error("Abort the protocol and ban the party {0}")]
52    AbortProtocolAndBanParty(u8),
53}
54
55impl From<MessageSendError> for SignError {
56    fn from(_err: MessageSendError) -> Self {
57        SignError::SendMessage
58    }
59}
60
61impl From<k256::ecdsa::Error> for SignError {
62    fn from(_err: k256::ecdsa::Error) -> Self {
63        Self::K256Error
64    }
65}
66
67impl From<Error> for SignError {
68    fn from(err: Error) -> Self {
69        match err {
70            Error::Abort(p) => SignError::AbortProtocol(p as _),
71            Error::Recv => SignError::MissingMessage,
72            Error::Send => SignError::SendMessage,
73            Error::InvalidMessage => SignError::InvalidMessage,
74        }
75    }
76}
