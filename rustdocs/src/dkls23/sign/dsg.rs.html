43/// Inner function for the pre-signature phase of the DSG protocol
44///
45/// This function implements the core logic of the pre-signature phase,
46/// where parties generate a pre-signature that can be used to sign any
47/// message later.
48///
49/// # Type Parameters
50///
51/// * `R`: Type implementing the `Relay` trait for message communication
52/// * `S`: Type implementing the `PreSignSetupMessage` trait for setup parameters
53///
54/// # Arguments
55///
56/// * `setup`: Setup parameters for the protocol
57/// * `seed`: Random seed for generating random values
58/// * `relay`: Message relay for communication between parties
59///
60/// # Returns
61///
62/// A `Result` containing either:
63/// * `Ok(PreSign)`: The pre-signature result
64/// * `Err(SignError)`: An error if the protocol fails
65async fn pre_signature_inner<R: Relay, S: PreSignSetupMessage>(
66    setup: &S,
67    seed: Seed,
68    relay: &mut FilteredMsgRelay<R>,
69) -> Result<PreSign, SignError> {
70    let mut rng = ChaCha20Rng::from_seed(seed);
71    let mut scheme = crate::proto::Scheme::new(&mut rng);
72
73    // For DKG part_id == part_idx.
74    //
75    // For DSG: party_idx is an index of the party in the setup messages.
76    //
77    // In the first message a party sends its part_id from Keyshare and
78    // its encryption public key
79    //
80    let my_party_id = setup.keyshare().party_id;
81    let my_party_idx = setup.participant_index();
82
83    let phi_i: Scalar = Scalar::generate_biased(&mut rng);
84    let r_i: Scalar = Scalar::generate_biased(&mut rng);
85    let blind_factor: [u8; 32] = rng.gen();
86
87    let big_r_i = ProjectivePoint::GENERATOR * r_i;
88
89    // TODO: Replace with SmallVec 2?
90    let mut commitments = vec![([0; 32], [0; 32]); setup.total_participants()];
91
92    commitments[my_party_idx].0 = rng.gen(); // generate SessionId
93    commitments[my_party_idx].1 =
94        hash_commitment_r_i(&commitments[my_party_idx].0, &big_r_i, &blind_factor);
95
96    relay
97        .send(SignedMessage::build(
98            &setup.msg_id(None, DSG_MSG_R1),
99            setup.message_ttl().as_secs() as _,
100            0,
101            setup.signer(),
102            |msg: &mut SignMsg1, _| {
103                msg.session_id = commitments[my_party_idx].0;
104                msg.commitment_r_i = commitments[my_party_idx].1;
105                msg.party_id = my_party_id;
106                msg.enc_pk = scheme.public_key().try_into().unwrap();
107            },
108        ))
109        .await?;
110
111    // vector of pairs (party_idx, party_id)
112    let mut party_idx_to_id_map = vec![(my_party_idx, my_party_id)];
113
114    Round::new(setup.total_participants() - 1, DSG_MSG_R1, relay)
115        .of_signed_messages(
116            setup,
117            SignError::AbortProtocol,
118            |msg: &SignMsg1, party_idx| {
119                party_idx_to_id_map.push((party_idx, msg.party_id));
120                commitments[party_idx] = (msg.session_id, msg.commitment_r_i);
121                scheme
122                    .receiver_public_key(party_idx, &msg.enc_pk)
123                    .map_err(|_| SignError::InvalidMessage)?;
124
125                Ok(())
126            },
127        )
128        .await?;
129
130    party_idx_to_id_map.sort_by_key(|&(_, pid)| pid);
131
132    // there is no party-id duplicates
133    if !party_idx_to_id_map.windows(2).all(|w| w[0] != w[1]) {
134        return Err(SignError::InvalidMessage);
135    }
136
137    // all party-id are in range
138    if !party_idx_to_id_map
139        .iter()
140        .all(|&(_, pid)| pid < setup.keyshare().total_parties)
141    {
142        return Err(SignError::InvalidMessage);
143    }
144
145    // IDX -> ID
146    let find_party_id = |idx: usize| {
147        party_idx_to_id_map
148            .iter()
149            .find_map(|&(i, p)| (i == idx).then_some(p))
150            .unwrap()
151    };
152
153    let final_session_id: [u8; 32] = commitments
154        .iter()
155        .fold(Sha256::new(), |hash, (sid, _)| hash.chain_update(sid))
156        .chain_update(setup.keyshare().final_session_id)
157        .finalize()
158        .into();
159
160    let digest_i: [u8; 32] = commitments
161        .iter()
162        .enumerate()
163        .fold(
164            Sha256::new().chain_update(DSG_LABEL),
165            |hash, (key, (sid, commitment))| {
166                hash.chain_update((key as u32).to_be_bytes())
167                    .chain_update(sid)
168                    .chain_update(commitment)
169            },
170        )
171        .chain_update(DIGEST_I_LABEL)
172        .finalize()
173        .into();
174
175    let mut to_send = vec![];
176
177    let mut mta_receivers = Pairs::from(
178        setup
179            .all_other_parties()
180            .map(|party_idx| {
181                let sender_id = find_party_id(party_idx);
182
183                let sid = mta_session_id(&final_session_id, sender_id, my_party_id);
184
185                let sender_ot_results = setup.keyshare().sender_seed(sender_id);
186
187                let mut enc_msg = EncryptedMessage::<SignMsg2>::new(
188                    &setup.msg_id(Some(party_idx), DSG_MSG_R2),
189                    setup.message_ttl().as_secs() as u32,
190                    0,
191                    0,
192                    &scheme,
193                );
194
195                let (msg2, _) = enc_msg.payload(&scheme);
196                msg2.final_session_id = final_session_id;
197
198                let (mta_receiver, chi_i_j) =
199                    RVOLEReceiver::new(sid, sender_ot_results, &mut msg2.mta_msg1, &mut rng);
200
201                to_send.push(
202                    enc_msg
203                        .encrypt(&mut scheme, party_idx)
204                        .ok_or(SignError::SendMessage)?,
205                );
206
207                Ok((party_idx, (mta_receiver, chi_i_j)))
208            })
209            .collect::<Result<Vec<_>, SignError>>()?,
210    );
211
212    for msg in to_send {
213        relay.feed(msg).await.map_err(|_| SignError::SendMessage)?;
214    }
215
216    relay.flush().await?;
217
218    let zeta_i = get_zeta_i(setup.keyshare(), &party_idx_to_id_map, &digest_i);
219
220    let coeff = if setup.keyshare().zero_ranks() {
221        get_lagrange_coeff(setup.keyshare(), &party_idx_to_id_map)
222    } else {
223        let betta_coeffs = get_birkhoff_coefficients(setup.keyshare(), &party_idx_to_id_map);
224
225        *betta_coeffs
226            .get(&(my_party_id as usize))
227            .expect("betta_i not found") // FIXME
228    };
229
230    let (additive_offset, derived_public_key) = setup
231        .keyshare()
232        .derive_with_offset(setup.chain_path())
233        .unwrap(); // FIXME: report error
234    let threshold_inv = Scalar::from(setup.total_participants() as u32)
235        .invert()
236        .unwrap(); // threshold > 0 so it has an invert
237    let additive_offset = additive_offset * threshold_inv;
238
239    let sk_i = coeff * setup.keyshare().s_i() + additive_offset + zeta_i;
240    let pk_i = ProjectivePoint::GENERATOR * sk_i;
241
242    let mut sender_additive_shares = vec![];
243
244    let mut round = Round::new(setup.total_participants() - 1, DSG_MSG_R2, relay);
245
246    while let Some((msg, party_idx, is_abort)) = round.recv().await? {
247        if is_abort {
248            check_abort(setup, &msg, party_idx, SignError::AbortProtocol)?;
249            round.put_back(&msg, ABORT_MESSAGE_TAG, party_idx);
250            continue;
251        }
252
253        let mut msg = Zeroizing::new(msg);
254        let msg2 = match EncryptedMessage::<SignMsg2>::decrypt(&mut msg, 0, &scheme, party_idx) {
255            Some((refs, _)) => refs,
256            _ => {
257                round.put_back(&msg, DSG_MSG_R2, party_idx);
258                continue;
259            }
260        };
261
262        // Check final_session_id
263        if msg2.final_session_id.ct_ne(&final_session_id).into() {
264            return Err(SignError::InvalidFinalSessionID);
265        }
266
267        let receiver_id = find_party_id(party_idx);
268
269        let sid = mta_session_id(&final_session_id, my_party_id, receiver_id);
270
271        let seed_ot_results = setup.keyshare().receiver_seed(receiver_id);
272
273        let mut enc_msg3 = EncryptedMessage::<SignMsg3>::new(
274            &setup.msg_id(Some(party_idx), DSG_MSG_R3),
275            setup.message_ttl().as_secs() as _,
276            0,
277            0,
278            &scheme,
279        );
280
281        let (msg3, _) = enc_msg3.payload(&scheme);
282
283        let [c_u, c_v] = RVOLESender::process(
284            &sid,
285            seed_ot_results,
286            &[r_i, sk_i],
287            &msg2.mta_msg1,
288            &mut msg3.mta_msg2,
289            &mut rng,
290        )
291        .map_err(|_| SignError::AbortProtocolAndBanParty(party_idx as u8))?;
292
293        let gamma_u = ProjectivePoint::GENERATOR * c_u;
294        let gamma_v = ProjectivePoint::GENERATOR * c_v;
295        let (_mta_receiver, chi_i_j) = mta_receivers.find_pair(party_idx);
296
297        let psi = phi_i - chi_i_j;
298
299        msg3.final_session_id = final_session_id;
300        msg3.digest_i = digest_i;
301        msg3.pk_i = encode_point(&pk_i);
302        msg3.big_r_i = encode_point(&big_r_i);
303        msg3.blind_factor = blind_factor;
304        msg3.gamma_v = encode_point(&gamma_v);
305        msg3.gamma_u = encode_point(&gamma_u);
306        msg3.psi = encode_scalar(&psi);
307
308        round
309            .relay
310            .send(
311                enc_msg3
312                    .encrypt(&mut scheme, party_idx)
313                    .ok_or(SignError::SendMessage)?,
314            )
315            .await?;
316
317        sender_additive_shares.push([c_u, c_v]);
318    }
319
320    let mut big_r_star = ProjectivePoint::IDENTITY;
321    let mut sum_pk_j = ProjectivePoint::IDENTITY;
322    let mut sum_psi_j_i = Scalar::ZERO;
323
324    let mut receiver_additive_shares = vec![];
325
326    let mut round = Round::new(setup.total_participants() - 1, DSG_MSG_R3, relay);
327
328    while let Some((msg, party_idx, is_abort)) = round.recv().await? {
329        if is_abort {
330            check_abort(setup, &msg, party_idx, SignError::AbortProtocol)?;
331            round.put_back(&msg, ABORT_MESSAGE_TAG, party_idx);
332            continue;
333        }
334
335        let mut msg = Zeroizing::new(msg);
336        let msg3 = match EncryptedMessage::<SignMsg3>::decrypt(&mut msg, 0, &scheme, party_idx) {
337            Some((refs, _)) => refs,
338            _ => {
339                round.put_back(&msg, DSG_MSG_R3, party_idx);
340                continue;
341            }
342        };
343
344        // Check final_session_id
345        if msg3.final_session_id != final_session_id {
346            return Err(SignError::InvalidFinalSessionID);
347        }
348
349        let (mta_receiver, chi_i_j) = mta_receivers.pop_pair(party_idx);
350
351        let [d_u, d_v] = mta_receiver
352            .process(&msg3.mta_msg2)
353            .map_err(|_| SignError::AbortProtocolAndBanParty(party_idx as u8))?;
354
355        let (sid_i, commitment) = &commitments[party_idx];
356
357        let big_r_j = decode_point(&msg3.big_r_i).ok_or(SignError::InvalidMessage)?;
358
359        if !verify_commitment_r_i(sid_i, &big_r_j, &msg3.blind_factor, commitment) {
360            return Err(SignError::InvalidCommitment);
361        }
362
363        if digest_i.ct_ne(&msg3.digest_i).into() {
364            return Err(SignError::InvalidDigest);
365        }
366
367        let pk_j = decode_point(&msg3.pk_i).ok_or(SignError::InvalidMessage)?;
368
369        big_r_star += big_r_j;
370        sum_pk_j += pk_j;
371        sum_psi_j_i += decode_scalar(&msg3.psi).ok_or(SignError::InvalidMessage)?;
372
373        let cond1 = (big_r_j * chi_i_j)
374            == (ProjectivePoint::GENERATOR * d_u
375                + decode_point(&msg3.gamma_u).ok_or(SignError::InvalidMessage)?);
376        if !cond1 {
377            return Err(SignError::AbortProtocolAndBanParty(party_idx as u8));
378        }
379
380        let cond2 = (pk_j * chi_i_j)
381            == (ProjectivePoint::GENERATOR * d_v
382                + decode_point(&msg3.gamma_v).ok_or(SignError::InvalidMessage)?);
383        if !cond2 {
384            return Err(SignError::AbortProtocolAndBanParty(party_idx as u8));
385        }
386
387        receiver_additive_shares.push([d_u, d_v]);
388    }
389
390    // new var
391    let big_r = big_r_star + big_r_i;
392    sum_pk_j += pk_i;
393
394    // Checks
395    if sum_pk_j != derived_public_key {
396        return Err(SignError::FailedCheck("Consistency check 3 failed"));
397    }
398
399    let mut sum_v = Scalar::ZERO;
400    let mut sum_u = Scalar::ZERO;
401
402    for i in 0..setup.total_participants() - 1 {
403        let sender_shares = &sender_additive_shares[i];
404        let receiver_shares = &receiver_additive_shares[i];
405        sum_u += sender_shares[0] + receiver_shares[0];
406        sum_v += sender_shares[1] + receiver_shares[1];
407    }
408
409    let r_point = big_r.to_affine();
410    let r_x = <Scalar as Reduce<U256>>::reduce_bytes(&r_point.x());
411    let phi_plus_sum_psi = phi_i + sum_psi_j_i;
412    let s_0 = r_x * (sk_i * phi_plus_sum_psi + sum_v);
413    let s_1 = r_i * phi_plus_sum_psi + sum_u;
414
415    let pre_sign_result = PreSign {
416        final_session_id,
417        public_key: encode_point(&derived_public_key),
418        s_0: encode_scalar(&s_0),
419        s_1: encode_scalar(&s_1),
420        phi_i: encode_scalar(&phi_i),
421        r: encode_point(&big_r),
422        party_id: my_party_id,
423    };
424
425    Ok(pre_sign_result)
426}

428/// Creates a partial signature from a pre-signature result
429///
430/// This function takes a pre-signature result and a message hash,
431/// and creates a partial signature that can be combined with other
432/// partial signatures to form the final signature.
433///
434/// # Arguments
435///
436/// * `pre_sign_result`: The pre-signature result from the pre-signature phase
437/// * `message_hash`: The hash of the message to be signed
438///
439/// # Returns
440///
441/// A `Result` containing either:
442/// * `Ok(PartialSignature)`: The partial signature
443/// * `Err(SignError)`: An error if the partial signature cannot be created
444fn create_partial_signature(
445    pre_sign_result: &PreSign,
446    message_hash: [u8; 32],
447) -> Result<PartialSignature, SignError> {
448    let m = Scalar::reduce(U256::from_be_slice(&message_hash));
449
450    let phi_i = decode_scalar(&pre_sign_result.phi_i).ok_or(SignError::InvalidPreSign)?;
451
452    let s_0 = decode_scalar(&pre_sign_result.s_0).ok_or(SignError::InvalidPreSign)?;
453
454    let s_0 = m * phi_i + s_0;
455
456    let s_1 = decode_scalar(&pre_sign_result.s_1).ok_or(SignError::InvalidPreSign)?;
457
458    let r = decode_point(&pre_sign_result.r).ok_or(SignError::InvalidPreSign)?;
459
460    let public_key = decode_point(&pre_sign_result.public_key).ok_or(SignError::InvalidPreSign)?;
461
462    Ok(PartialSignature {
463        final_session_id: pre_sign_result.final_session_id,
464        public_key,
465        message_hash,
466        s_0,
467        s_1,
468        r,
469    })
470}

472/// Combines partial signatures into a final signature
473///
474/// This function takes a collection of partial signatures and combines
475/// them to produce the final ECDSA signature and recovery ID.
476///
477/// # Arguments
478///
479/// * `partial_signatures`: A slice of partial signatures to combine
480///
481/// # Returns
482///
483/// A `Result` containing either:
484/// * `Ok((Signature, RecoveryId))`: The final signature and recovery ID
485/// * `Err(SignError)`: An error if the signatures cannot be combined
486fn combine_partial_signature(
487    partial_signatures: &[PartialSignature],
488) -> Result<(Signature, RecoveryId), SignError> {
489    let p0 = &partial_signatures[0];
490
491    let mut check = Choice::from(0);
492
493    let mut sum_s_0 = p0.s_0;
494    let mut sum_s_1 = p0.s_1;
495
496    for pn in &partial_signatures[1..] {
497        check |= pn.final_session_id.ct_ne(&p0.final_session_id);
498        check |= pn.public_key.ct_ne(&p0.public_key);
499        check |= pn.r.ct_ne(&p0.r);
500        check |= pn.message_hash.ct_ne(&p0.message_hash);
501
502        sum_s_0 += pn.s_0;
503        sum_s_1 += pn.s_1;
504    }
505
506    if check.into() {
507        return Err(SignError::FailedCheck("Invalid list of partial signatures"));
508    }
509
510    let r = p0.r.to_affine();
511
512    let is_y_odd: bool = r.y_is_odd().into();
513
514    let r_x = <Scalar as Reduce<U256>>::reduce_bytes(&r.x());
515    let is_x_reduced = r_x.to_repr() != r.x();
516    let recid = RecoveryId::new(is_y_odd, is_x_reduced);
517
518    let sum_s_1_inv = sum_s_1.invert().unwrap();
519    let s = sum_s_0 * sum_s_1_inv;
520
521    let is_y_odd = recid.is_y_odd() ^ bool::from(s.is_high());
522    let recid = RecoveryId::new(is_y_odd, recid.is_x_reduced());
523
524    let sign = Signature::from_scalars(r_x, s)?;
525    let sign = sign.normalize_s().unwrap_or(sign);
526
527    VerifyingKey::from_affine(p0.public_key.to_affine())?
528        .verify_prehash(&p0.message_hash, &sign)?;
529
530    Ok((sign, recid))
531}

533/// Main entry point for the DSG protocol
534///
535/// This function executes the complete DSG protocol, including both
536/// the pre-signature and finish phases.
537///
538/// # Type Parameters
539///
540/// * `R`: Type implementing the `Relay` trait for message communication
541/// * `S`: Type implementing the `SignSetupMessage` trait for setup parameters
542///
543/// # Arguments
544///
545/// * `setup`: Setup parameters for the protocol
546/// * `seed`: Random seed for generating random values
547/// * `relay`: Message relay for communication between parties
548///
549/// # Returns
550///
551/// A `Result` containing either:
552/// * `Ok((Signature, RecoveryId))`: The final signature and recovery ID
553/// * `Err(SignError)`: An error if the protocol fails
554pub async fn run<R: Relay, S: SignSetupMessage>(
555    setup: S,
556    seed: Seed,
557    relay: R,
558) -> Result<(Signature, RecoveryId), SignError> {
559    let abort_msg = create_abort_message(&setup);
560    let mut relay = FilteredMsgRelay::new(relay);
561
562    relay.ask_messages(&setup, ABORT_MESSAGE_TAG, false).await?;
563    relay.ask_messages(&setup, DSG_MSG_R1, false).await?;
564    relay.ask_messages(&setup, DSG_MSG_R2, true).await?;
565    relay.ask_messages(&setup, DSG_MSG_R3, true).await?;
566    relay.ask_messages(&setup, DSG_MSG_R4, false).await?;
567
568    let result = match run_inner(setup, seed, &mut relay).await {
569        Ok(sign) => Ok(sign),
570        Err(SignError::AbortProtocol(p)) => Err(SignError::AbortProtocol(p)),
571        Err(SignError::SendMessage) => Err(SignError::SendMessage),
572        Err(err) => {
573            // ignore error of sending abort message
574            let _ = relay.send(abort_msg).await;
575            Err(err)
576        }
577    };
578
579    let _ = relay.close().await;
580
581    result
582}

584/// Inner function for the main DSG protocol execution
585///
586/// This function implements the core logic of the DSG protocol,
587/// handling both the pre-signature and finish phases.
588///
589/// # Type Parameters
590///
591/// * `R`: Type implementing the `Relay` trait for message communication
592/// * `S`: Type implementing the `SignSetupMessage` trait for setup parameters
593///
594/// # Arguments
595///
596/// * `setup`: Setup parameters for the protocol
597/// * `seed`: Random seed for generating random values
598/// * `relay`: Message relay for communication between parties
599///
600/// # Returns
601///
602/// A `Result` containing either:
603/// * `Ok((Signature, RecoveryId))`: The final signature and recovery ID
604/// * `Err(SignError)`: An error if the protocol fails
605async fn run_inner<R: Relay, S: SignSetupMessage>(
606    setup: S,
607    seed: Seed,
608    relay: &mut FilteredMsgRelay<R>,
609) -> Result<(Signature, RecoveryId), SignError> {
610    let t = setup.total_participants();
611
612    let pre_signature_result = pre_signature_inner(&setup, seed, relay).await?;
613
614    let msg_hash = setup.message_hash();
615
616    run_final(&setup, relay, t, msg_hash, &pre_signature_result).await
617}

619/// Executes the pre-signature phase of the DSG protocol
620///
621/// This function runs only the pre-signature phase of the protocol,
622/// producing a pre-signature that can be used later to sign messages.
623///
624/// # Type Parameters
625///
626/// * `R`: Type implementing the `Relay` trait for message communication
627/// * `S`: Type implementing the `PreSignSetupMessage` trait for setup parameters
628///
629/// # Arguments
630///
631/// * `setup`: Setup parameters for the protocol
632/// * `seed`: Random seed for generating random values
633/// * `relay`: Message relay for communication between parties
634///
635/// # Returns
636///
637/// A `Result` containing either:
638/// * `Ok(PreSign)`: The pre-signature result
639/// * `Err(SignError)`: An error if the protocol fails
640pub async fn pre_signature<R: Relay, S: PreSignSetupMessage>(
641    setup: S,
642    seed: Seed,
643    relay: R,
644) -> Result<PreSign, SignError> {
645    let abort_msg = create_abort_message(&setup);
646    let mut relay = FilteredMsgRelay::new(relay);
647
648    relay.ask_messages(&setup, ABORT_MESSAGE_TAG, false).await?;
649    relay.ask_messages(&setup, DSG_MSG_R1, false).await?;
650    relay.ask_messages(&setup, DSG_MSG_R2, true).await?;
651    relay.ask_messages(&setup, DSG_MSG_R3, true).await?;
652
653    let result = match pre_signature_inner(&setup, seed, &mut relay).await {
654        Ok(result) => Ok(result),
655        Err(SignError::AbortProtocol(p)) => Err(SignError::AbortProtocol(p)),
656        Err(SignError::SendMessage) => Err(SignError::SendMessage),
657        Err(err) => {
658            relay.send(abort_msg).await?;
659            Err(err)
660        }
661    };
662
663    let _ = relay.close().await;
664
665    result
666}

668/// Executes the finish phase of the DSG protocol
669///
670/// This function runs the finish phase of the protocol, using a
671/// pre-signature to generate the final signature for a message.
672///
673/// # Type Parameters
674///
675/// * `R`: Type implementing the `Relay` trait for message communication
676/// * `S`: Type implementing the `FinalSignSetupMessage` trait for setup parameters
677///
678/// # Arguments
679///
680/// * `setup`: Setup parameters for the protocol
681/// * `relay`: Message relay for communication between parties
682///
683/// # Returns
684///
685/// A `Result` containing either:
686/// * `Ok((Signature, RecoveryId))`: The final signature and recovery ID
687/// * `Err(SignError)`: An error if the protocol fails
688pub async fn finish<R: Relay, S: FinalSignSetupMessage>(
689    setup: S,
690    relay: R,
691) -> Result<(Signature, RecoveryId), SignError> {
692    let pre_signature_result = setup.pre_signature();
693    let msg_hash = setup.message_hash();
694    let mut relay = FilteredMsgRelay::new(relay);
695
696    relay.ask_messages(&setup, ABORT_MESSAGE_TAG, false).await?;
697    relay.ask_messages(&setup, DSG_MSG_R4, false).await?;
698
699    let result = run_final(
700        &setup,
701        &mut relay,
702        setup.total_participants(),
703        msg_hash,
704        pre_signature_result,
705    )
706    .await;
707
708    let _ = relay.close().await;
709
710    result
711}

713/// Inner function for the finish phase of the DSG protocol
714///
715/// This function implements the core logic of the finish phase,
716/// where parties use a pre-signature to generate the final signature.
717///
718/// # Type Parameters
719///
720/// * `R`: Type implementing the `Relay` trait for message communication
721/// * `S`: Type implementing the `ProtocolParticipant` trait for participant information
722///
723/// # Arguments
724///
725/// * `setup`: Setup parameters for the protocol
726/// * `relay`: Message relay for communication between parties
727/// * `t`: Threshold value for the signature
728/// * `msg_hash`: Hash of the message to be signed
729/// * `pre_signature_result`: The pre-signature result from the pre-signature phase
730///
731/// # Returns
732///
733/// A `Result` containing either:
734/// * `Ok((Signature, RecoveryId))`: The final signature and recovery ID
735/// * `Err(SignError)`: An error if the protocol fails
736async fn run_final<R: Relay, S: ProtocolParticipant>(
737    setup: &S,
738    relay: &mut FilteredMsgRelay<R>,
739    t: usize,
740    msg_hash: [u8; 32],
741    pre_signature_result: &PreSign,
742) -> Result<(Signature, RecoveryId), SignError> {
743    let public_key = decode_point(&pre_signature_result.public_key).unwrap();
744    let r = decode_point(&pre_signature_result.r).unwrap();
745
746    let partial_signature = create_partial_signature(pre_signature_result, msg_hash)?;
747
748    relay
749        .send(SignedMessage::build(
750            &setup.msg_id(None, DSG_MSG_R4),
751            setup.message_ttl().as_secs() as _,
752            0,
753            setup.signer(),
754            |msg4: &mut SignMsg4, _| {
755                msg4.session_id = partial_signature.final_session_id;
756                msg4.s_0 = encode_scalar(&partial_signature.s_0);
757                msg4.s_1 = encode_scalar(&partial_signature.s_1);
758            },
759        ))
760        .await?;
761
762    let mut partial_signatures: Vec<PartialSignature> = Vec::with_capacity(t);
763
764    partial_signatures.push(partial_signature);
765
766    Round::new(setup.total_participants() - 1, DSG_MSG_R4, relay)
767        .of_signed_messages(
768            setup,
769            SignError::AbortProtocol,
770            |msg: &SignMsg4, _party_idx| {
771                partial_signatures.push(PartialSignature {
772                    final_session_id: msg.session_id,
773                    public_key,
774                    message_hash: msg_hash,
775                    s_0: decode_scalar(&msg.s_0).ok_or(SignError::InvalidMessage)?,
776                    s_1: decode_scalar(&msg.s_1).ok_or(SignError::InvalidMessage)?,
777                    r,
778                });
779
780                Ok(())
781            },
782        )
783        .await?;
784
785    combine_partial_signature(&partial_signatures)
786}

788/// Computes the hash of a commitment value
789///
790/// This function computes the hash of a commitment value using the
791/// session ID, R point, and blind factor.
792///
793/// # Arguments
794///
795/// * `session_id`: The session identifier
796/// * `big_r_i`: The R point value
797/// * `blind_factor`: The blind factor value
798///
799/// # Returns
800///
801/// A 32-byte array containing the hash of the commitment
802fn hash_commitment_r_i(
803    session_id: &[u8],
804    big_r_i: &ProjectivePoint,
805    blind_factor: &[u8; 32],
806) -> [u8; 32] {
807    let mut hasher = Sha256::new();
808    hasher.update(DSG_LABEL);
809    hasher.update(session_id.as_ref());
810    hasher.update(big_r_i.to_bytes());
811    hasher.update(blind_factor);
812    hasher.update(COMMITMENT_LABEL);
813
814    hasher.finalize().into()
815}

817/// Computes the zeta_i value for a party
818///
819/// This function computes the zeta_i value used in the signature
820/// generation process.
821///
822/// # Arguments
823///
824/// * `keyshare`: The key share of the party
825/// * `party_id_list`: List of party IDs participating in the protocol
826/// * `sig_id`: The signature identifier
827///
828/// # Returns
829///
830/// The computed zeta_i scalar value
831fn get_zeta_i(keyshare: &Keyshare, party_id_list: &[(usize, u8)], sig_id: &[u8]) -> Scalar {
832    let mut sum_p_0 = Scalar::ZERO;
833    for &(_, p_0_party) in party_id_list {
834        if p_0_party >= keyshare.party_id {
835            continue;
836        }
837
838        let seed_j_i = keyshare.each(p_0_party).zeta_seed;
839
840        let mut hasher = Sha256::new();
841        hasher.update(DSG_LABEL);
842        hasher.update(seed_j_i);
843        hasher.update(sig_id);
844        hasher.update(PAIRWISE_RANDOMIZATION_LABEL);
845        sum_p_0 += Scalar::reduce(U256::from_be_slice(&hasher.finalize()));
846    }
847
848    let mut sum_p_1 = Scalar::ZERO;
849    for &(_, p_1_party) in party_id_list {
850        if p_1_party <= keyshare.party_id {
851            continue;
852        }
853
854        let seed_i_j = keyshare.each(p_1_party - 1).zeta_seed;
855
856        let mut hasher = Sha256::new();
857        hasher.update(DSG_LABEL);
858        hasher.update(seed_i_j);
859        hasher.update(sig_id);
860        hasher.update(PAIRWISE_RANDOMIZATION_LABEL);
861        sum_p_1 += Scalar::reduce(U256::from_be_slice(&hasher.finalize()));
862    }
863
864    sum_p_0 - sum_p_1
865}

867/// Computes the Birkhoff coefficients for the protocol
868///
869/// This function computes the Birkhoff coefficients used in the
870/// signature generation process.
871///
872/// # Arguments
873///
874/// * `keyshare`: The key share of the party
875/// * `sign_party_ids`: List of party IDs participating in the protocol
876///
877/// # Returns
878///
879/// A map of party indices to their corresponding Birkhoff coefficients
880fn get_birkhoff_coefficients(
881    keyshare: &Keyshare,
882    sign_party_ids: &[(usize, u8)],
883) -> HashMap<usize, Scalar> {
884    let params = sign_party_ids
885        .iter()
886        .map(|&(_, pid)| (keyshare.get_x_i(pid), keyshare.get_rank(pid) as usize))
887        .collect::<Vec<_>>();
888
889    let betta_vec = birkhoff_coeffs::<Secp256k1>(&params);
890
891    sign_party_ids
892        .iter()
893        .zip(betta_vec.iter())
894        .map(|((_, pid), w_i)| (*pid as usize, *w_i))
895        .collect::<HashMap<_, _>>()
896}

898/// Computes the Lagrange coefficient for a party
899///
900/// This function computes the Lagrange coefficient used in the
901/// signature generation process.
902///
903/// # Arguments
904///
905/// * `keyshare`: The key share of the party
906/// * `sign_party_ids`: List of party IDs participating in the protocol
907///
908/// # Returns
909///
910/// The computed Lagrange coefficient
911fn get_lagrange_coeff(keyshare: &Keyshare, sign_party_ids: &[(usize, u8)]) -> Scalar {
912    let mut coeff = Scalar::from(1u64);
913    let pid = keyshare.party_id;
914    let x_i = &keyshare.get_x_i(pid) as &Scalar;
915
916    for &(_, party_id) in sign_party_ids {
917        let x_j = &keyshare.get_x_i(party_id) as &Scalar;
918        if x_i.ct_ne(x_j).into() {
919            let sub = x_j - x_i;
920            coeff *= x_j * &sub.invert().unwrap();
921        }
922    }
923
924    coeff
925}

927/// Computes a list of Lagrange coefficients
928///
929/// This function computes a list of Lagrange coefficients for a set
930/// of party points.
931///
932/// # Type Parameters
933///
934/// * `K`: Type of the key function
935/// * `T`: Type of the party points
936///
937/// # Arguments
938///
939/// * `party_points`: List of party points
940/// * `k`: Function to extract the key from a party point
941///
942/// # Returns
943///
944/// An iterator over the computed Lagrange coefficients
945pub(crate) fn get_lagrange_coeff_list<'a, K, T>(
946    party_points: &'a [T],
947    k: K,
948) -> impl Iterator<Item = Scalar> + 'a
949where
950    K: Fn(&T) -> &NonZeroScalar + 'a,
951{
952    party_points.iter().map(move |x_i| {
953        let x_i = k(x_i);
954        let mut coeff = Scalar::ONE;
955        for x_j in party_points {
956            let x_j = k(x_j);
957            if x_i.ct_ne(x_j).into() {
958                let sub = x_j.sub(x_i);
959                // SAFETY: Invert is safe because we check x_j != x_i, so sub is not zero.
960                coeff *= x_j.as_ref() * &sub.invert().unwrap();
961            }
962        }
963        coeff
964    })
965}

967/// Verifies a commitment value
968///
969/// This function verifies that a commitment value matches the expected
970/// hash of the session ID, R point, and blind factor.
971///
972/// # Arguments
973///
974/// * `sid`: The session identifier
975/// * `big_r_i`: The R point value
976/// * `blind_factor`: The blind factor value
977/// * `commitment`: The commitment value to verify
978///
979/// # Returns
980///
981/// `true` if the commitment is valid, `false` otherwise
982fn verify_commitment_r_i(
983    sid: &[u8],
984    big_r_i: &ProjectivePoint,
985    blind_factor: &[u8; 32],
986    commitment: &[u8],
987) -> bool {
988    let compare_commitment = hash_commitment_r_i(sid, big_r_i, blind_factor);
989
990    commitment.ct_eq(&compare_commitment).into()
991}

993/// Generates a session ID for the MtA protocol
994///
995/// This function generates a unique session ID for the MtA protocol
996/// based on the final session ID and the sender/receiver IDs.
997///
998/// # Arguments
999///
1000/// * `final_session_id`: The final session identifier
1001/// * `sender_id`: The ID of the sender party
1002/// * `receiver_id`: The ID of the receiver party
1003///
1004/// # Returns
1005///
1006/// A 32-byte array containing the generated session ID
1007fn mta_session_id(final_session_id: &[u8], sender_id: u8, receiver_id: u8) -> [u8; 32] {
1008    let mut h = Sha256::new();
1009    h.update(DSG_LABEL);
1010    h.update(final_session_id);
1011    h.update(b"sender");
1012    h.update([sender_id]);
1013    h.update(b"receiver");
1014    h.update([receiver_id]);
1015    h.update(PAIRWISE_MTA_LABEL);
1016    h.finalize().into()
1017}
