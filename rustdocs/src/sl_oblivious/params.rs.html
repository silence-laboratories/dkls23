4pub mod consts {
5    // Bits on underlying Scalar
6    pub const KAPPA: usize = 256;
7
8    // Computational security parameter, fixed to /lambda_c = 256
9    // 256 OT seeds each 256-bit
10    pub const LAMBDA_C: usize = 256;
11
12    pub const LAMBDA_S: usize = 128;
13
14    pub const S: usize = 128; // 16 bytes == 128 bits
15
16    pub const L_BATCH: usize = 2;
17    pub const RHO: usize = 1;
18    pub const SOFT_SPOKEN_K: usize = 4;
19
20    // size of u8 array to hold LAMBDA_C bits.
21    pub const LAMBDA_C_BYTES: usize = LAMBDA_C / 8;
22
23    pub const KAPPA_BYTES: usize = KAPPA >> 3;
24
25    pub const S_BYTES: usize = S >> 3;
26
27    pub const L: usize = KAPPA + 2 * LAMBDA_S; // L is divisible by S
28    pub const L_BYTES: usize = L >> 3;
29
30    pub const L_PRIME: usize = L + S;
31    pub const L_PRIME_BYTES: usize = L_PRIME >> 3;
32
33    pub const SOFT_SPOKEN_M: usize = L / S;
34
35    pub const L_BATCH_PLUS_RHO: usize = L_BATCH + RHO; // should be equal to OT_WIDTH
36    pub const OT_WIDTH: usize = L_BATCH_PLUS_RHO;
37
38    pub const SOFT_SPOKEN_Q: usize = 1 << SOFT_SPOKEN_K;
39
40    pub const LAMBDA_C_DIV_SOFT_SPOKEN_K: usize = LAMBDA_C / SOFT_SPOKEN_K;
41}
