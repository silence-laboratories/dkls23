4///     Multiplies `a` and `b` in the finite field of order 2^128
5///     modulo the irreducible polynomial f(x) = 1 + x + x^2 + x^7 + x^128
6
7///     https://link.springer.com/book/10.1007/b97644
8///     multiplication part: Algorithm 2.34, "Right-to-left comb method for polynomial multiplication"
9///     reduction part: variant of the idea of Figure 2.9
10pub fn binary_field_multiply_gf_2_128(
11    a: &[u8; 16],
12    b_data: &[u8; 16],
13) -> [u8; 16] {
14    const W: usize = 8;
15    const T: usize = 16;
16
17    let mut c = [0u8; T * 2];
18    let mut b = [0u8; T + 1];
19
20    b[..16].copy_from_slice(b_data);
21
22    for k in 0..W {
23        for j in 0..T {
24            //let mask = if (a[j] >> k) & 0x01 == 1 { 0xFF } else { 0x00 };
25            let mask = -((a[j] >> k & 0x01) as i8) as u8;
26            for i in 0..T + 1 {
27                c[j + i] ^= b[i] & mask;
28            }
29        }
30        for i in (1..=T).rev() {
31            b[i] = (b[i] << 1) | (b[i - 1] >> 7);
32        }
33        b[0] <<= 1
34    }
35    for i in (T..=2 * T - 1).rev() {
36        c[i - 16] ^= c[i];
37        c[i - 16] ^= c[i] << 1;
38        c[i - 15] ^= c[i] >> 7;
39        c[i - 16] ^= c[i] << 2;
40        c[i - 15] ^= c[i] >> 6;
41        c[i - 16] ^= c[i] << 7;
42        c[i - 15] ^= c[i] >> 1;
43    }
44
45    c[..16].try_into().unwrap()
46}

49mod tests {
50    // Test based on fermat's little theorem
51    #[test]
52    fn test_mul_128() {
53        for _ in 0..10 {
54            let rand_num = rand::random::<[u8; 16]>();
55            let mut temp = rand_num;
56
57            for _ in 0..128 {
58                temp = super::binary_field_multiply_gf_2_128(&temp, &temp);
59            }
60
61            assert_eq!(temp, rand_num);
62        }
63    }
